% Logo (PDF preferred)
\newcommand{\FSLogo}{%
  \begingroup
  \IfFileExists{assets/fs_logo_blue.pdf}{%

  }{%
    \changefont{\textcolor{gray}{FS}}%
  }%
  \endgroup
}

% Header text is injected from YAML (Pandoc-expanded)
\newcommand{\FSHeaderText}{}

\pagestyle{fancy}
\fancyhf{}
\fancyhead[R]{\FSLogo}
\fancyhead[C]{\changefont{\texttt{\textcolor{gray}{\FSHeaderText}}}}

% Make the first page (plain pagestyle) use the same header/footer as fancy
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyhead[R]{\FSLogo}%
  \fancyhead[C]{\changefont{\texttt{\textcolor{gray}{\FSHeaderText}}}}%
  \fancyfoot[C]{\changefont\textbf{\thepage}}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0pt}%
}
